  <script>
    /* ==========================================================================
       Mock Data & Application State
       ========================================================================== */
    const state = {
      mode: 'summary', // 'summary', 'detail', 'register'
      selectedTreeId: null,
      tempMarker: null,
      selectedLatLng: null,
      uploadedPhotoBase64: null,
      trees: [
        {
          id: 'TR-001',

          lng: 139.7090,

          circ: 65,

          photo: null
        },
        {
          id: 'TR-002',
          species: 'ソメイヨシノ',
          lat: 35.6842,
          lng: 139.7112,
          height: 8.5,
          circ: 35,
          vigor: 3, // 少し悪い
          risk: 'II', // 被害中
          photo: null
        },
        {

          species: 'イチョウ',
          lat: 35.6828,
          lng: 139.7080,
          height: 18.0,
          circ: 80,
          vigor: 4, // 悪い
          risk: 'IV', // 倒伏危険性高
          photo: null
        }
      ]
    };

    const vigorLabels = {
      1: '良い',
      2: '普通',
      3: '少し悪い',
      4: '悪い',
      5: '枯死'
    };

    const riskLabels = {
      'I': 'I: 被害無〜小',
      'II': 'II: 被害中',
      'III': 'III: 倒伏危険性低',
      'IV': 'IV: 倒伏危険性高'
    };

    /* ==========================================================================
       DOM Elements
       ========================================================================== */
    const summaryView = document.getElementById('summaryView');
    const detailView = document.getElementById('detailView');
    const registerView = document.getElementById('registerView');
    
    const triggerRegisterBtn = document.getElementById('triggerRegisterBtn');
    const backToSummaryBtn = document.getElementById('backToSummaryBtn');
    const cancelRegisterBtn = document.getElementById('cancelRegisterBtn');
    const submitBtn = document.getElementById('submitBtn');
    
    const treeListContainer = document.getElementById('treeList');
    
    // Form elements
    const registrationForm = document.getElementById('registrationForm');
    const inputPhoto = document.getElementById('inputPhoto');
    const previewImg = document.getElementById('previewImg');
    const photoPlaceholder = document.getElementById('photoPlaceholder');
    const inputLat = document.getElementById('inputLat');
    const inputLng = document.getElementById('inputLng');
    const inputSpecies = document.getElementById('inputSpecies');
    const inputHeight = document.getElementById('inputHeight');
    const inputCirc = document.getElementById('inputCirc');
    const selectVigor = document.getElementById('selectVigor');
    const selectRisk = document.getElementById('selectRisk');

    // Details elements
    const detailTreePhoto = document.getElementById('detailTreePhoto');
    const detailTreeFallbackGraphic = document.getElementById('detailTreeFallbackGraphic');
    const detailTreeName = document.getElementById('detailTreeName');
    const detailTreeId = document.getElementById('detailTreeId');
    const detailRiskBadge = document.getElementById('detailRiskBadge');
    const detailTreeHeight = document.getElementById('detailTreeHeight');
    const detailTreeCirc = document.getElementById('detailTreeCirc');
    const detailTreeVigor = document.getElementById('detailTreeVigor');
    const detailTreeLatLng = document.getElementById('detailTreeLatLng');

    /* ==========================================================================
       Map & Leaflet Marker setups
       ========================================================================== */
    let map = null;
    let markers = {};

    function initMap() {
      // Shinjuku Gyoen focus
      map = L.map('map', {
        zoomControl: true,
        attributionControl: true
      }).setView([35.6850, 139.7100], 16);

      // CartoDB Dark Matter tile layer
      L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
        attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> &copy; <a href="https://carto.com/attributions">CARTO</a>',
        subdomains: 'abcd',
        maxZoom: 20
      }).addTo(map);

      // Move zoom control
      map.zoomControl.setPosition('topright');

      // Map click handler (for position fetching in register mode)
      map.on('click', (e) => {
        if (state.mode !== 'register') return;
        const { lat, lng } = e.latlng;
        state.selectedLatLng = e.latlng;
        
        inputLat.value = lat.toFixed(6);
        inputLng.value = lng.toFixed(6);
        
        // Render or update temporary marker
        if (state.tempMarker) {
          state.tempMarker.setLatLng(e.latlng);
        } else {
          const tempIcon = createCustomMarkerIcon('temp');
          state.tempMarker = L.marker(e.latlng, { icon: tempIcon }).addTo(map);
        }
      });
    }

    function createCustomMarkerIcon(riskClass) {
      return L.divIcon({
        className: 'custom-marker-wrapper',
        html: `
          <div class="marker-pulse ${riskClass}"></div>
          <div class="marker-dot ${riskClass}"></div>
        `,
        iconSize: [24, 24],
        iconAnchor: [12, 12]
      });
    }

    function renderMarkers() {
      // Clear existing markers from map
      Object.keys(markers).forEach(id => {
        map.removeLayer(markers[id]);
      });
      markers = {};

      // Draw all markers from state
      state.trees.forEach(tree => {
        const riskClass = 'risk-' + tree.risk.toLowerCase();
        const icon = createCustomMarkerIcon(riskClass);
        const marker = L.marker([tree.lat, tree.lng], { icon: icon }).addTo(map);
        
        markers[tree.id] = marker;

        // Custom marker click listener
        marker.on('click', () => {
          selectTree(tree.id);
          // Pan map offset slightly to accommodate sidebar width
          const offsetLng = window.innerWidth > 768 ? -0.001 : 0;
          map.setView([tree.lat, tree.lng + offsetLng], 17, { animate: true, duration: 0.6 });
        });
      });
    }

    /* ==========================================================================
       Mode Navigation State Machine
       ========================================================================== */
    function switchMode(newMode) {
      state.mode = newMode;
      
      // Hide all panels
      summaryView.classList.add('hidden');
      detailView.classList.add('hidden');
      registerView.classList.add('hidden');

      // Map highlight resets
      Object.keys(markers).forEach(id => {
        const el = markers[id].getElement();
        if (el) el.classList.remove('active');
      });

      // Temporary marker cleanup if exiting register mode
      if (newMode !== 'register' && state.tempMarker) {
        map.removeLayer(state.tempMarker);
        state.tempMarker = null;
        state.selectedLatLng = null;
      }

      // Display selected panel & perform initialization
      if (newMode === 'summary') {
        summaryView.classList.remove('hidden');
        populateTreeList();
        calculateStats();
      } else if (newMode === 'detail') {
        detailView.classList.remove('hidden');
      } else if (newMode === 'register') {
        registerView.classList.remove('hidden');
        resetForm();
      }
    }

    /* ==========================================================================
       Summary Mode: Population & Stats calculation
       ========================================================================== */
    function populateTreeList() {
      treeListContainer.innerHTML = '';
      state.trees.forEach(tree => {
        const item = document.createElement('div');
        item.className = 'tree-item';
        item.dataset.id = tree.id;
        
        const riskClass = 'badge-' + tree.risk.toLowerCase();
        item.innerHTML = `
          <div class="tree-item-left">
            <span class="tree-item-name">${tree.species}</span>
            <span class="tree-item-id">${tree.id}</span>
          </div>
          <span class="risk-badge ${riskClass}">RANK ${tree.risk}</span>
        `;

        item.addEventListener('click', () => {
          selectTree(tree.id);
          const offsetLng = window.innerWidth > 768 ? -0.001 : 0;
          map.setView([tree.lat, tree.lng + offsetLng], 17, { animate: true, duration: 0.6 });
        });

        treeListContainer.appendChild(item);
      });
    }

    function calculateStats() {
      const total = state.trees.length;
      const dangerous = state.trees.filter(t => t.risk === 'III' || t.risk === 'IV').length;
      const attentionRate = total > 0 ? Math.round((state.trees.filter(t => t.risk !== 'I').length / total) * 100) : 0;

      document.getElementById('statTotal').innerText = total;
      document.getElementById('statDanger').innerText = dangerous;
      document.getElementById('statAttentionRate').innerText = `${attentionRate}%`;
    }

    /* ==========================================================================
       Detail Mode: Fill fields & Render
       ========================================================================== */
    function selectTree(treeId) {
      state.selectedTreeId = treeId;
      const tree = state.trees.find(t => t.id === treeId);
      if (!tree) return;

      switchMode('detail');

      // Highlight active marker on map
      const marker = markers[treeId];
      if (marker) {
        const el = marker.getElement();
        if (el) el.classList.add('active');
      }

      // Populate details
      detailTreeId.innerText = tree.id;
      detailTreeName.innerText = tree.species;
      detailTreeHeight.innerText = tree.height.toFixed(1);
      detailTreeCirc.innerText = tree.circ;
      detailTreeLatLng.innerText = `${tree.lat.toFixed(5)}, ${tree.lng.toFixed(5)}`;
      
      // Vigor class & label mapping
      detailTreeVigor.innerText = vigorLabels[tree.vigor];
      detailTreeVigor.className = 'detail-spec-value vigor-text vigor-' + tree.vigor;

      // Risk class mapping
      detailRiskBadge.className = 'risk-badge badge-' + tree.risk.toLowerCase();
      detailRiskBadge.innerText = 'RANK ' + tree.risk;

      // Photo rendering (Fallback check)
      if (tree.photo) {
        detailTreePhoto.src = tree.photo;
        detailTreePhoto.classList.remove('hidden');
        detailTreeFallbackGraphic.classList.add('hidden');
      } else {
        detailTreePhoto.classList.add('hidden');
        detailTreeFallbackGraphic.classList.remove('hidden');
      }
    }

    /* ==========================================================================
       Register Mode: Photo Upload & Submit Card logic
       ========================================================================== */
    function resetForm() {
      registrationForm.reset();
      previewImg.src = '';
      previewImg.classList.add('hidden');
      photoPlaceholder.classList.remove('hidden');
      state.uploadedPhotoBase64 = null;
      inputLat.value = '';
      inputLng.value = '';
    }

    // Photo input change (Base64 file reader preview)
    inputPhoto.addEventListener('change', (e) => {
      const file = e.target.files[0];
      if (file) {
        const reader = new FileReader();
        reader.onload = (event) => {
          state.uploadedPhotoBase64 = event.target.result;
          previewImg.src = state.uploadedPhotoBase64;
          previewImg.classList.remove('hidden');
          photoPlaceholder.classList.add('hidden');
        };
        reader.readAsDataURL(file);
      }
    });

    // Form submission
    registrationForm.addEventListener('submit', (e) => {
      e.preventDefault();

      const lat = parseFloat(inputLat.value);
      const lng = parseFloat(inputLng.value);
      
      if (isNaN(lat) || isNaN(lng)) {
        alert('地図をクリックして、位置情報（緯度・経度）を取得してください。');
        return;
      }

      // Generate ID
      const nextIdNum = state.trees.length + 1;
      const nextIdStr = 'TR-' + String(nextIdNum).padStart(3, '0');

      // Create new Tree item
      const newTree = {
        id: nextIdStr,
        species: inputSpecies.value.trim(),
        lat: lat,
        lng: lng,
        height: parseFloat(inputHeight.value),
        circ: parseInt(inputCirc.value, 10),
        vigor: parseInt(selectVigor.value, 10),
        risk: selectRisk.value,
        photo: state.uploadedPhotoBase64
      };

      // Push to state & Re-render Map pins
      state.trees.push(newTree);
      renderMarkers();

      // Clear & switch
      alert(`樹木カルテ ${nextIdStr} を正常に登録しました。`);
      switchMode('summary');
    });

    /* ==========================================================================
       Setup Listeners & Start
       ========================================================================== */
    triggerRegisterBtn.addEventListener('click', () => switchMode('register'));
    backToSummaryBtn.addEventListener('click', () => switchMode('summary'));
    cancelRegisterBtn.addEventListener('click', () => switchMode('summary'));

    window.addEventListener('DOMContentLoaded', () => {
      initMap();
      renderMarkers();
      switchMode('summary');
    });
